Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
.\ultralytics_env\Scripts\activate

CLEAR STATE
sudo modprobe uvcvideo
RESET ALL CAMERA V4L2 SETTINGS
v4l2-ctl -d /dev/video0 --set-ctrl=white_balance_automatic=1
v4l2-ctl -d /dev/video0 --set-ctrl=auto_exposure=3
v4l2-ctl -d /dev/video0 --set-ctrl=exposure_time_absolute=100
v4l2-ctl -d /dev/video0 --set-ctrl=gain=0
v4l2-ctl -d /dev/video0 --list-ctrls

FORCE RESOLUTION AND MJPG
v4l2-ctl -d /dev/video0 --set-fmt-video=width=640,height=480,pixelformat=MJPG

# CLAUDES "NUCLEAR RESET"
udevadm info /dev/video0 | grep DEVPATH
# Then unbind/rebind the USB device (replace 'X-X.X' with your actual path)
echo "X-X.X" | sudo tee /sys/bus/usb/drivers/usb/unbind
sleep 2
echo "X-X.X" | sudo tee /sys/bus/usb/drivers/usb/bind
